207 glm::vec3 CollisionDetector::GetContactBetweenEdges(const std::pair<glm::vec3, glm::vec3>& edgeA,
208  const std::pair<glm::vec3, glm::vec3>& edgeB)
209 {
210  glm::vec3 e1 = edgeA.second - edgeA.first;
211  glm::vec3 e2 = edgeB.second - edgeB.first;
212  glm::vec3 r = edgeB.first - edgeA.first;
213  glm::vec3 normal = glm::cross(e1, e2);
214  glm::vec3 normal1 = glm::cross(e1, normal);
215  glm::vec3 normal2 = glm::cross(e2, normal);
216  glm::vec3 p1 = edgeA.first + (glm::dot(r, normal2)/glm::dot(e1, normal2)) * e1;
217  glm::vec3 p2 = edgeB.first + (glm::dot(-r, normal1)/glm::dot(e2, normal1)) * e2;
218  return p1 + 0.5f * (p2 - p1);

221 float CollisionDetector::GetMinDistanceBetweenEdges(const std::pair<glm::vec3, glm::vec3>& edgeA,
222  const std::pair<glm::vec3, glm::vec3>& edgeB)
223 {
224  glm::vec3 e1 = edgeA.second - edgeA.first;
225  glm::vec3 e2 = edgeB.second - edgeB.first;
226  glm::vec3 w = edgeA.first - edgeB.first;
227  float a = glm::dot(e1, e1);
228  float b = glm::dot(e1, e2);
229  float c = glm::dot(e2, e2);
230  float d = glm::dot(e1, w);
231  float e = glm::dot(e2, w);
232  float D = a*c - b*b;
233  float sc, sNominator, sDenominator = D;
234  float tc, tNominator, tDenominator = D;
235 
236  if (D < 0.0005f)
237  {
238  sNominator = 0.f;
239  sDenominator = 1.f;
240  tNominator = e;
241  tDenominator = c;
242  }
243  else
244  {
245  sNominator = b*e - c*d;
246  tNominator = a*e - b*d;
247  if (sNominator < 0.f) // closest point is closer to s=0
248  {
249  sNominator = 0.f;
250  tNominator = e;
251  tDenominator = c;
252  }
253  else if (sNominator > sDenominator) // closest point is closer to s=1
254  {
255  sNominator = sDenominator;
256  tNominator = e + b;
257  tDenominator = c;
258  }
259  }
260 
261  if (tNominator < 0.f)
262  {
263  tNominator = 0.f;
264  if (-d < 0.f)
265  {
266  sNominator = 0.f;
267  }
268  else if (-d > a)
269  {
270  sNominator = sDenominator;
271  }
272  else
273  {
274  sNominator = -d;
275  sDenominator = a;
276  }
277  }
278  else if (tNominator > tDenominator)
279  {
280  tNominator = tDenominator;
281  if ((-d + b) < 0.f)
282  {
283  sNominator = 0.f;
284  }
285  else if ((-d + b) > a)
286  {
287  sNominator = sDenominator;
288  }
289  else
290  {
291  sNominator = -d + b;
292  sDenominator = a;
293  }
294  }
295  sc = (abs(sNominator) < this->tolerance ? 0.0 : sNominator / sDenominator);
296  tc = (abs(tNominator) < this->tolerance ? 0.0 : tNominator / tDenominator);
297  glm::vec3 result = w + sc * e1 - tc * e2;
298  // dont need actual distance
299  return glm::length2(result);
300 }

355 std::vector<glm::vec3> CollisionDetector::Clip(std::vector<glm::vec3>& points, std::vector<std::pair<glm::vec3, glm::vec3>>& planes)
356 {
357  // Used Sutherland-Hodgman for the clipping
358  // https://en.wikipedia.org/wiki/Sutherland%E2%80%93Hodgman_algorithm
359  std::vector<glm::vec3> output = points;
360  for (int j = 0; j < planes.size(); j++)
361  {
362  std::vector<glm::vec3> input = output;
363  output.clear();
364  std::pair<glm::vec3, glm::vec3> plane = planes[j];
365  float planeOffset = glm::dot(plane.first, plane.second);
366  for (int i = 0; i < input.size(); i++)
367  {
368  // compute intersection points
369  glm::vec3 v1 = input[i];
370  glm::vec3 v2 = input[(i+1) % input.size()];
371  glm::vec3 intersectionPoint = this->IntersectLinePlane(v1, v2, plane);
372  if (glm::dot(plane.first, v2) < planeOffset)
373  {
374  if (glm::dot(plane.first, v1) > planeOffset)
375  {
376  output.push_back(intersectionPoint);
377  }
378  output.push_back(v2);
379  }
380  else if (glm::dot(plane.first, v1) < planeOffset)
381  {
382  output.push_back(intersectionPoint);
383  }
384  }
385  }
386  return output;
387 }
