148  /* / Q
149  / |
150  / |
151  /\ |
152  M ================= N
153  P
154  */
155  int index = 0;
156  glm::vec3 edgeFirst = points[edge.first];
157  glm::vec3 edgeSecond = points[edge.second];
158  for (int i = 0; i < pointIndices.size(); ++i)
159  {
160  if (pointIndices[i] != edge.first && pointIndices[i] != edge.second)
161  {
162  index = i;
163  break;
164  }
165  }
166  glm::vec3 indexPoint = points[pointIndices[index]];
167  glm::vec3 MQ = indexPoint - edgeFirst;
168  glm::vec3 MN = glm::normalize(edgeSecond - edgeFirst);
169  glm::vec3 PQ = glm::normalize(MQ - glm::dot(MQ, MN) * MN);
170  for (int i = 0; i < pointIndices.size(); i++)
171  {
172  if (pointIndices[i] == edge.first || pointIndices[i] == edge.second)
173  continue;
174  glm::vec3 currPoint = points[pointIndices[i]];
175  glm::vec3 currDir = glm::normalize(currPoint - edgeFirst);
176  if (glm::dot(currDir, PQ) < 0.f)
177  {
178  return false;
179  }
180  }
181  return true;
