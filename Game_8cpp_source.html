131 void Game::InitScene(std::string filename, std::vector<std::unique_ptr<Entity>>& entities)
132 {
133  unsigned int textureID = this->renderingSystem.CreateTexture("resources/DiffuseColor_Texture.png");
134  tinyxml2::XMLDocument document;
135  // TODO : check for the extension and report error if different from .dae
136  tinyxml2::XMLError error = document.LoadFile(filename.c_str());
137  if (error != 0)
138  return;
139  // the document represents the "whole" file so we need to qu
140  // is always Collada
141  tinyxml2::XMLElement* collada = document.FirstChildElement("COLLADA");
142 
143  // parse visual scenes for the world transforms.
144  std::unordered_map<std::string, std::shared_ptr<InstanceGeometry>> instanceGeometries;
145  tinyxml2::XMLElement* libraryVisualScenes = collada->FirstChildElement("library_visual_scenes");
146  instanceGeometries = Loader::ParseVisualScenesStatic(libraryVisualScenes);
147 
148  // parse geometries
149  // one object - > many colliders
150  std::unordered_map<std::string, std::vector<std::shared_ptr<Collider>> > objectToColliders;
151  std::unordered_map<std::string, std::shared_ptr<Geometry>> geometry;
152  tinyxml2::XMLElement* libraryGeometries = collada->FirstChildElement("library_geometries");
153  geometry = Loader::ParseGeometry(libraryGeometries);
154  // first iteration to get hitboxes
155  for (std::unordered_map<std::string, std::shared_ptr<Geometry>>::iterator it = geometry.begin(); it != geometry.end(); it++)
156  {
157  // Three scenarios
158  // is hitbox
159  // is not hitbox
160  // is terrain
161  bool isHitbox = it->first.find("_hitbox") != it->first.npos;
162  if (isHitbox)
163  {
164  std::vector<glm::vec3> points;
165  int hitboxIndex = it->first.find("_hitbox");
166  std::string objectName = it->first.substr(0, hitboxIndex);
167  // take thhe vertices and make a collider.
168  for (int i = 0; i < it->second->vertices.size(); i += 3)
169  {
170  float x = it->second->vertices[i];
171  float y = it->second->vertices[i + 1];
172  float z = it->second->vertices[i + 2];
173  glm::vec3 point = glm::vec3(x,y,z);
174  points.push_back(point);
175  }
176  DynamicType type = DynamicType::Static;
177  if (objectName == "player")
178  type = DynamicType::Dynamic;
179  std::shared_ptr<Collider> collider = ColliderBuilder::Build(0, type, points);
180  objectToColliders[objectName].push_back(collider);
181  }
182  }
183  std::vector<float> bufferData;
184  glm::mat4 worldTransform;
185  // second iteration to create game entities
186  for (std::unordered_map<std::string, std::shared_ptr<Geometry>>::iterator it = geometry.begin(); it != geometry.end(); it++)
187  {
188  bool isHitbox = it->first.find("_hitbox") != it->first.npos;
189  if (isHitbox)
190  continue;
191 
192  std::shared_ptr<Geometry> current = it->second;
193  std::unique_ptr<Entity> entity = std::make_unique<Entity>(this->CreateEntityID());
194 
195  bufferData = Loader::BuildBufferData(current);
196  worldTransform = instanceGeometries[it->first]->matrix;
197  std::pair<unsigned int, unsigned int> buffers = RenderingSystem::BufferData(bufferData.data(), bufferData.size() * sizeof(float), false);
198  glm::vec3 scale;
199  glm::quat rotation;
200  glm::vec3 translation;
201  glm::vec3 skew;
202  glm::vec4 perspective;
203  glm::decompose(worldTransform, scale, rotation, translation, skew, perspective);
204  rotation = glm::conjugate(rotation);
205  // RenderingComponent
206  // we need to have
207  // VAO, VBO and Texture loaded.
208  std::unique_ptr<RenderingComponent> renderingComponent = std::make_unique<RenderingComponent>(buffers.first, buffers.second, bufferData.size() / 3, textureID, ShaderType::NormalShader);
209  // PhysicsComponent
210  DynamicType type = DynamicType::Static;
211  float mass = 1.0f;
212  if (it->first == "player")
213  type = DynamicType::Dynamic;
214  else
215  mass = 1000.f;
216  printVector(translation, "INIT TRANSLATION");
217  std::unique_ptr<PhysicsComponent> physicsComponent = std::make_unique<PhysicsComponent>(1.f, translation, rotation, glm::mat3(1.f), type);
218  // assign colliders to component and insert into grid
219  physicsComponent->colliders = objectToColliders[it->first];
220  for (int k = 0;k < physicsComponent->colliders.size(); k++)
221  {
222  physicsComponent->colliders[k]->entityID = entity->id;
223  }
224  this->physicsSystem.Insert(physicsComponent->colliders);
225  std::unique_ptr<TransformComponent> transformComponent = std::make_unique<TransformComponent>(translation, rotation);
226 
227  // Entity
228  entity->AddComponent(std::move(renderingComponent));
229  entity->AddComponent(std::move(physicsComponent));
230  entity->AddComponent(std::move(transformComponent));
231  if (it->first == "player")
232  {
233  this->playerID = entity->id;
234  std::unique_ptr<InputComponent> inputComponent = std::make_unique<InputComponent>();
235  entity->AddComponent(std::move(inputComponent));
236  }
237  entities.push_back(std::move(entity));
238  }
239  // Push back an entity
240 }

292 void Game::Dispatch()
293 {
294  // clear previous frame messages
295  for (int i = 0; i < System::SystemEnd; i++)
296  {
297  this->systemToMessage[i].clear();
298  }
299  // dispatch messages
300  for (int i = 0;i < this->globalQueue.size(); i++)
301  {
302  Message message = this->globalQueue[i];
303  std::vector<System> interestedSystems = this->messageToSystem[message.type];
304  for (int j = 0; j < interestedSystems.size(); j++)
305  {
306  System system = interestedSystems[j];
307  this->systemToMessage[system].push_back(message);
308  }
309  }
310  this->globalQueue.clear();
311 }
